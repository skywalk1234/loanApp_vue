在贷款产品页面loan.vue刷新完accessToken之后，向后端发送请求获取贷款产品，请求示例代码如下
~~~javascript
var myHeaders = new Headers();
myHeaders.append("Authorization", accessToken);
//accessToken从localStorage中获取
var requestOptions = {
   method: 'GET',
   headers: myHeaders,
   redirect: 'follow'
};

fetch("http://115.190.40.44:45444/loan/option", requestOptions)
   .then(response => response.text())
   .then(result => console.log(result))
   .catch(error => console.log('error', error));
~~~
